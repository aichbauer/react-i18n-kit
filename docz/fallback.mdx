```js
import React from "react";
import { useI18n } from "react-i18n-kit";

import { data } from "./i18n";

const TextHook = () => {
  const { i18n } = useI18n(data, {
    lang: 'de',
    fallback: 'en',
  });

  return (
    <>
      <div>{i18n.text}</div>
      <div>{i18n.text2}</div>
    </>
  );
};

export { TextHook };
```
