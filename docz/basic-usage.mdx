```jsx
import React from 'react';
import { withI18n } from 'react-i18n-kit';

import { data } from './i18n';

/*
  if language is:
    - en:
      props.i18n.text: "Hello World!"
    - de:
      props.i18n.text: "Hallo Welt!"
*/
const Text = (props) => (
  <div>
    {props.i18n.text}
  </div>
);

const TextI18n = withI18n(Text, data);

export { TextI18n as Text };
```
