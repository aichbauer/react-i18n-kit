```jsx
import React from "react";
import { useI18n } from "react-i18n-kit";

import { data } from "./i18n";

/*
  if language is:
    - en:
      i18n.text: "Hello World!"
    - de:
      i18n.text: "Hallo Welt!"
*/
const Text = () => {
  const { i18n } = useI18n(data)

  return (
    <div>{i18n.text}</div>
  );
};

export { Text };
```
